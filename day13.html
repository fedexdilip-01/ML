💾 1. Model Serialization – Saving & Loading ML Models
📦 Pickle vs Joblib: Which to use?
📐 Pickle

Python's native serialization. Works for any Python object.

⚠️ Security: Only load trusted pickles!

📊 Joblib

Optimized for large NumPy arrays (scikit-learn models). 2-10x faster than pickle for ML models.

⚙️ Best practice

Use joblib for scikit-learn models, pickle for custom objects.

# Train and save an Iris classifier
from sklearn.datasets import load_iris
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
import joblib
import pickle

# Train model
iris = load_iris()
X_train, X_test, y_train, y_test = train_test_split(iris.data, iris.target, test_size=0.2, random_state=42)
model = RandomForestClassifier(n_estimators=100, random_state=42)
model.fit(X_train, y_train)
print(f"Model accuracy: {model.score(X_test, y_test):.4f}")

# Save with joblib (recommended)
joblib.dump(model, 'iris_model.joblib')
print("✅ Model saved as 'iris_model.joblib'")

# Save with pickle
with open('iris_model.pkl', 'wb') as f:
    pickle.dump(model, f)
print("✅ Model saved as 'iris_model.pkl'")

# Load model
loaded_model = joblib.load('iris_model.joblib')
print(f"Loaded model accuracy: {loaded_model.score(X_test, y_test):.4f}")
💡 Pro tip: Save preprocessors (scalers, encoders) alongside models! Deploy them together.
🌐 2. FastAPI – Build REST API for Model Inference
⚡ Async, auto-documentation, type validation
📐 Key features

Pydantic data validation, automatic OpenAPI (Swagger) docs, async support, high performance (on par with Node/Go).

🎯 Endpoint design

POST /predict – receives features, returns predictions.

⚙️ Response format

JSON with predictions, probabilities, model version.

# app.py - Complete FastAPI service for Iris classification
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, Field
from typing import List
import joblib
import numpy as np
import uvicorn

# Define request/response models
class IrisFeatures(BaseModel):
    sepal_length: float = Field(..., ge=0, le=10, description="Sepal length in cm")
    sepal_width: float = Field(..., ge=0, le=10, description="Sepal width in cm")
    petal_length: float = Field(..., ge=0, le=10, description="Petal length in cm")
    petal_width: float = Field(..., ge=0, le=10, description="Petal width in cm")
    
    class Config:
        schema_extra = {
            "example": {
                "sepal_length": 5.1,
                "sepal_width": 3.5,
                "petal_length": 1.4,
                "petal_width": 0.2
            }
        }

class PredictionResponse(BaseModel):
    prediction: int
    species: str
    probabilities: List[float]
    model_version: str = "1.0.0"

# Initialize FastAPI
app = FastAPI(
    title="Iris Classifier API",
    description="Predict iris flower species from sepal/petal measurements",
    version="1.0.0"
)

# Load model at startup
model = joblib.load('iris_model.joblib')
iris_classes = ['setosa', 'versicolor', 'virginica']

@app.get("/")
def root():
    return {"message": "Iris Classifier API", "endpoints": ["/predict", "/health"]}

@app.get("/health")
def health():
    return {"status": "healthy", "model_loaded": True}

@app.post("/predict", response_model=PredictionResponse)
def predict(features: IrisFeatures):
    try:
        # Convert to numpy array
        input_array = np.array([[
            features.sepal_length,
            features.sepal_width,
            features.petal_length,
            features.petal_width
        ]])
        
        # Predict
        prediction = model.predict(input_array)[0]
        probabilities = model.predict_proba(input_array)[0].tolist()
        
        return PredictionResponse(
            prediction=int(prediction),
            species=iris_classes[prediction],
            probabilities=probabilities
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

# Run with: uvicorn app:app --reload --host 0.0.0.0 --port 8000
if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)
📌 Run the API: `uvicorn app:app --reload` → Visit `http://localhost:8000/docs` for interactive Swagger UI!

🐳 3. Docker – Containerize the API
📦 Dockerfile: Package app + dependencies + model
📐 Why Docker?

"Works on my machine" → guaranteed consistency across environments. Easy cloud deployment (AWS ECS, GCP Run, Azure).

🎯 Steps

Write Dockerfile → Build image → Run container.

# Dockerfile
FROM python:3.10-slim

WORKDIR /app

# Copy requirements and install
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy application code and model
COPY app.py .
COPY iris_model.joblib .

# Expose port
EXPOSE 8000

# Run the API
CMD ["uvicorn", "app:app", "--host", "0.0.0.0", "--port", "8000"]
# requirements.txt fastapi==0.104.1 uvicorn==0.24.0 scikit-learn==1.3.0 joblib==1.3.2 numpy==1.24.3 pydantic==2.4.2 
# Build Docker image
docker build -t iris-classifier .

# Run container
docker run -d -p 8000:8000 --name iris-api iris-classifier

# Test container
curl http://localhost:8000/health

# Stop container
docker stop iris-api

# Remove container
docker rm iris-api
🐳 Pro tip: Use multi-stage builds for smaller images, add health checks, and use environment variables for configuration.

# model.py - Training script
import joblib
from sklearn.datasets import load_iris
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split

def train_and_save():
    iris = load_iris()
    X_train, X_test, y_train, y_test = train_test_split(iris.data, iris.target, test_size=0.2, random_state=42)
    model = RandomForestClassifier(n_estimators=100, random_state=42)
    model.fit(X_train, y_train)
    joblib.dump(model, 'iris_model.joblib')
    print(f"Model saved. Test accuracy: {model.score(X_test, y_test):.4f}")

if __name__ == "__main__":
    train_and_save()

# docker-compose.yml (optional for multi-service)
version: '3.8'
services:
  api:
    build: .
    ports:
      - "8000:8000"
    environment:
      - MODEL_PATH=/app/iris_model.joblib
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
